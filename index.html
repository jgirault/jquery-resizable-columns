<script src="/path/to/jquery.js"></script>

<table class="table table-bordered" data-resizable-columns-id="demo-table">
<thead>
<tr>
<th data-resizable-column-id="#">#</th>
<th data-resizable-column-id="first_name">First Name</th>
<th data-resizable-column-id="last_name">Last Name</th>
<th data-resizable-column-id="username" data-noresize>Username</th>
</tr>
</thead>
<tbody>
...
</tbody>
</table>

<script>
$(function(){
  $('table[data-resizable-columns-id=demo-table]').resizableColumns({
    store: window.store
  });
});
</script>

<script src="/path/to/jquery.js"></script>
<script src="/path/to/store.js"></script>
<script src="/path/to/jquery.resizableColumns.js"></script>

<table class="table table-bordered" style="width: auto" data-resizable-columns-id="demo-table-pixels">
<thead>
  <tr>
    <th data-resizable-column-id="#">#</th>
    <th data-resizable-column-id="first_name">First Name</th>
    <th data-resizable-column-id="last_name">Last Name</th>
    <th data-resizable-column-id="username" data-noresize>Username</th>
  </tr>
</thead>
<tbody>
  ...
</tbody>
</table>

<script>
  $(function(){
      $('table[data-resizable-columns-id=demo-table-pixels]').resizableColumns({
        store: window.store,
        usePixels: true
      });
    });
</script>
